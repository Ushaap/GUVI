var request= new XMLHttpRequest();
request.open('GET','https://restcountries.eu/rest/v2/all',true);
request.send();
request.onload=function(){
    var data=JSON.parse(this.response);
    for(var i in data){
        var key=data[i].name;
        var val=data[i].flag;
        console.log(key+' '+val)
    }
    request.send()}
  
request.onload()
